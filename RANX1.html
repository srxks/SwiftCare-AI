function calculateCriticalityScore(hr, spo2, status) {
let score = 0;
// Heart rate scoring
if (hr > 150 || hr < 50) score += 40;
else if (hr > 120 || hr < 60) score += 20;
// SpO2 scoring
if (spo2 < 90) score += 40;
else if (spo2 < 94) score += 20;
// Status weighting
if (status === 'critical') score += 30;
else if (status === 'urgent') score += 15;
return Math.min(100, score);
}

function generateBP(status) {
if (status === 'critical') return `${Math.floor(Math.random() * 60) + 160}/${Math.floor(Math.random() * 40) + 90}`;
if (status === 'urgent') return `${Math.floor(Math.random() * 40) + 130}/${Math.floor(Math.random() * 30) + 75}`;
return `${Math.floor(Math.random() * 30) + 110}/${Math.floor(Math.random() * 20) + 65}`;
}

return room.number;

return 'Pending';
}

function generateAIRecommendations(patient) {
const container = document.getElementById('aiRecommendations');
container.innerHTML = '';
const recommendations = [];
// Generate recommendations based on vitals
if (patient.spo2 < 92) {
recommendations.push("Administer supplemental oxygen immediately");
}
if (patient.heartRate > 140) {
recommendations.push("Monitor for tachycardia complications - consider ECG");
}
if (patient.heartRate < 55) {
recommendations.push("Check for bradycardia causes - monitor cardiac output");
}
if (patient.temperature > 38.5) {
recommendations.push("Administer antipyretics and monitor for infection");
}
if (parseInt(patient.bp.split('/')[0]) > 160) {
recommendations.push("Monitor blood pressure - consider antihypertensives");
}
if (patient.respRate > 24) {
recommendations.push("Assess respiratory distress - check airway patency");
}
// Status-based recommendations
if (patient.status === 'critical') {
recommendations.push("Prepare trauma bay and emergency team");
recommendations.push("Notify senior physician immediately");
recommendations.push("Initiate advanced cardiac life support protocol");
} else if (patient.status === 'urgent') {
recommendations.push("Priority examination required within 15 minutes");
recommendations.push("Prepare intravenous access and monitoring");
}
// Condition-specific recommendations
if (patient.diagnosis.some(d => d.includes('Diabetes'))) {
recommendations.push("Check blood glucose levels immediately");
}
if (patient.diagnosis.some(d => d.includes('Asthma') || d.includes('COPD'))) {
recommendations.push("Have bronchodilators ready for administration");
}
if (patient.allergies.some(a => a.includes('Penicillin'))) {
recommendations.push("‚ö†Ô∏è ALLERGY ALERT: Avoid penicillin-based antibiotics");
}
// If no specific recommendations, add general ones
if (recommendations.length === 0) {
recommendations.push("Continue monitoring, patient appears stable");
recommendations.push("Maintain current treatment protocol");
}
// Limit to 6 recommendations
recommendations.slice(0, 6).forEach(rec => {
const li = document.createElement('li');
li.innerHTML = `<i class="fas fa-chevron-right" style="color: var(--primary); margin-right: 8px;"></i>${rec}`;
container.appendChild(li);
});
}

function startRealTimeUpdates() {
// Update patient vitals every 5 seconds
setInterval(() => {
patients.forEach(patient => {
// Simulate realistic vital changes
const change = Math.random() * 4 - 2; // -2 to +2
if (patient.status === 'critical') {
patient.heartRate += Math.random() * 6 - 3;
patient.spo2 += Math.random() * 2 - 1;
} else if (patient.status === 'urgent') {
patient.heartRate += Math.random() * 4 - 2;
patient.spo2 += Math.random() * 1 - 0.5;
} else {
patient.heartRate += Math.random() * 2 - 1;
patient.spo2 += Math.random() * 0.5 - 0.25;
}
// Keep vitals within realistic bounds
patient.heartRate = Math.max(40, Math.min(180, patient.heartRate));
patient.spo2 = Math.max(70, Math.min(100, patient.spo2));
// Update criticality score
patient.criticalityScore = calculateCriticalityScore(
Math.round(patient.heartRate),
Math.round(patient.spo2),
patient.status
);
// Update ETA (decrease by 1 minute every 10 seconds)
if (Math.random() > 0.7) {
const currentETA = parseInt(patient.eta);
if (currentETA > 1) {
patient.eta = `${currentETA - 1} min`;
}
}
});
renderPatients();
updatePatientCount();
// Update network stats
updateNetworkStats();
}, 5000);

// Update rooms periodically
setInterval(() => {
// Randomly change some room statuses
rooms.forEach(room => {
if (Math.random() > 0.95) {
const statuses = ['available', 'occupied', 'reserved'];
room.status = statuses[Math.floor(Math.random() * 3)];
}
});
renderRooms();
}, 10000);
}
